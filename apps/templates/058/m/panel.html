  6. Relevant grid code 
        var fields="Name,Color";
    m.fields="_Form,"+fields+",Submit_date,Submitted_by,_Delete";
    //-------------------------------------
    m.cell_render=function(records,I,field,td){
        switch(field){
            case 'Color':
                var h="<div style='height:20px; width:32px; background-color:"+records[I].Data[field]+"'></div>"
                td.html(h);
                break;
        }
    }
